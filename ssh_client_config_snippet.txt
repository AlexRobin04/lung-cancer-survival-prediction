# 以下内容请粘贴到你「本机电脑」的 ~/.ssh/config（没有则新建）
# Windows: C:\Users\你的用户名\.ssh\config
# macOS/Linux: ~/.ssh/config

Host *
  ServerAliveInterval 30
  ServerAliveCountMax 3
  TCPKeepAlive yes

# 若只想对某台服务器生效，可把 Host * 改成：
# Host 121.41.39.63
#   ServerAliveInterval 30
#   ServerAliveCountMax 3
#   TCPKeepAlive yes
